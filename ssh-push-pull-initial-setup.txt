#This section to be used to set up SSH connection from local to remote for GIT and GITHUB

#Install Git for MacOS
brew install git

#Verify git version after installation
git --version

#Configure GIT
git config --global user.name "user.name"
git config --global user.email "user.email"

#Verify git configuration
git config --global --list

#Generate SSH key 
ssh-keygen -t ed25519 -C "your_email@example.com"

#accept configuration and press enter twice to setup the keygen
/Users/yourusername/.ssh/id_ed25519

#Display pub.key

#Start SSH agent 
eval "$(ssh-agent -s)"

#Add public key

git init --> intialises the repo
git add . --> adds the untracked file to track by the git 
git commit -m "Initial commit" --> command to commit the tracked changes to the github remote location
git branch -M main      # rename branch to main, renames the local branch to main, needs to be done initially, as a standard set by github
git remote add origin git@github.com:howudooinng/repo-name.git --> this command helps point to the main branch of the repo, add your repo name which was created on the github.
git push -u origin main --> this helps pushes the changes to remote repo for the first time, after this this setup is not required to be done
